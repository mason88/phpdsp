172  /**
173  * Performs math operation for two waves.
174  * Binary because it operates on 2 operands, and not binary math.
175  *
176  * @param Wave $other instance of the other wave object
177  * @param string $op operation to perform (name of NumArray method)
178  *
179  * @return Wave
180  */
181  protected function binary_op($other, $op)
182  {
183  if (! $other) return($this);
184 
185  if ($this->framerate != $other->framerate)
186  trigger_error('Framerates for the two waves do not match.', E_USER_WARNING);
187 
188  // make an array of times that covers both waves
189  $start = min($this->start(), $other->start());
190  $end = max($this->end(), $other->end());
191  $n = intval(round(($end - $start) * $this->framerate)) + 1;
192  $ys = NumPHP::zeros($n);
193  $ts = NumPHP::arange(0, $n - 1)->div($this->framerate)->add($start);
194 
195  foreach(array($this, $other) as $wave) {
196  $i = $this->find_index_ts($ts);
197 
198  // make sure the arrays line up reasonably well
199  $diff = $ts->get($i)->getData() - $wave->start();
200  $dt = 1 / $wave->framerate;
201  if (($diff / $dt) > 0.1)
202  trigger_error("Can't operate these waveforms; their time arrays don't line up.", E_USER_WARNING);
203 
204  $j = $i + $wave->count();
205  $segment = $ys->get("$i:$j")->$op($wave->ys);
206  $ys->set("$i:$j", $segment);
207  }
208 
209  return new Wave($ys, $ts, $this->framerate);
210  }

244  /**
245  * Tapers the amplitude at the beginning and end of the signal.
246  * Tapers either the given duration of time or the given fraction of the total duration, whichever is less.
247  *
248  * @param float $denom fraction of the segment to taper
249  * @param float $duration duration of the taper in seconds
250  *
251  * @return $this
252  */
253  public function apodize($denom=20, $duration=0.1)
254  {
255  // a fixed fraction of the segment
256  $n = $this->count();
257  $k1 = intval(floor($n / $denom));
258 
259  // a fixed duration of time
260  $k2 = intval($duration * $this->framerate);
261 
262  $k = min($k1, $k2);
263 
264  $w1 = NumPHP::linspace(0, 1, $k);
265  $w2 = NumPHP::ones($n - (2 * $k));
266  $w3 = NumPHP::linspace(1, 0, $k);
267 
268  $window = new NumArray(array_merge($w1->getData(), $w2->getData(), $w3->getData()));
269  $this->ys->mult($window);
270 
271  return $this;
272  }

316  /**
317  * Computes the DCT of this wave using DCT-IV algorithm.
318  *
319  * @return DCT
320  */
321  public function make_dct()
322  {
323  $N = $this->count();
324 
325  $ts = NumPHP::arange(0, $N - 1)->add(0.5)->div($N);
326  $fs = NumPHP::arange(0, $N - 1)->add(0.5)->div(2);
327  $ts->outer($fs); // $ts is now args
328 
329  $ts->mult(2 * pi())->apply('cos'); // $ts is now M
330  $ts->dot($this->ys)->div(2); // $ts is now amps
331 
332  return new Dct($fs, $ts, $this->framerate);
333  }
